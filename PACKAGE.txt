CREATE OR REPLACE PACKAGE PACK1
IS
  CURSOR C1 IS SELECT * FROM EMP;
  PROCEDURE DISP;
  NO NUMBER;
END PACK1;
/



CREATE OR REPLACE PACKAGE BODY PACK1	
IS
  FUNCTION DISP_REC  --DEFINING A PRIVATE FUNCTION
  RETURN NUMBER
  IS
  CNT NUMBER;
  BEGIN
  SELECT COUNT(*) INTO CNT FROM EMP;
  RETURN CNT;
  END DISP_REC;

  PROCEDURE DISP
  IS
  BEGIN
  FOR MROW IN C1
  LOOP
  DBMS_OUTPUT.PUT_LINE(MROW.ENAME);
  END LOOP;

  NO:=DISP_REC; --CALLING THE PRIVATE FUNCTION FROM PROCEDURE
   DBMS_OUTPUT.PUT_LINE('NO OF ROWS ARE:'||NO);
END DISP;
END PACK1;
/
